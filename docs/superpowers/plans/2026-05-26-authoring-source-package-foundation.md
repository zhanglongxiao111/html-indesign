# 作者源码包基础设施 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 建立作者源码包基础设施，把当前架构汇报 fixture 从单个手写 `deck.html` 迁移为 `deck.config.json + styles/ + pages/`，并继续生成现有转换入口 `deck.html`。

**Architecture:** 新增 `src/authoring/` 作为作者源码包层，负责读取配置、校验页面片段、组装生成 `deck.html`。现有 `src/paged-html/` 仍只消费组装后的单文件 HTML，不直接理解多文件源码包。`scripts/lint-authoring.js` 增加 `--package` 入口，但保持原有 `<deck.html>` 用法兼容。

**Tech Stack:** Node.js CommonJS、Cheerio、现有 `node --test`、现有 Playwright 浏览器快照、现有 `npm run lint:authoring` 和 `npm run e2e:indesign`。

## Scope

本计划实现：

- `deck.config.json` v1 schema 的最小读取和校验。
- 作者源码包组装器：`deck.config.json + styles/*.css + pages/*.html -> deck.html`。
- CLI：`npm run assemble:authoring -- --package <deck.config.json>`。
- `lint:authoring` 支持 `--package <deck.config.json>`。
- 当前 `test/fixtures/e2e/architecture-report/deck.html` 迁移为源码包生成物。

本计划不实现：

- InDesign 标签中的 `sourcePackage/sourceFile/sourceNode` 写入。
- 反向导出 `author/` 源码包。
- 复杂网格推断。
- 自动把任意旧 HTML 拆成源码包。

这些内容进入下一张 plan，避免把源码包基础设施和反向链路耦合在一起。

Create `src/authoring/source-package.js`:

```js
const fs = require('fs');
const path = require('path');

const GENERATED_COMMENT = '<!-- Generated by html-indesign. Edit pages/*.html and styles/*.css, not this file. -->';

function readAuthorPackage(configPath) {
  if (!configPath) throw new Error('AUTHOR_PACKAGE_CONFIG_MISSING: config path is required');
  const resolvedConfigPath = path.resolve(configPath);
  if (!fs.existsSync(resolvedConfigPath)) {
    throw new Error(`AUTHOR_PACKAGE_CONFIG_MISSING: ${resolvedConfigPath}`);
  }

  const rootDir = path.dirname(resolvedConfigPath);
  const config = JSON.parse(fs.readFileSync(resolvedConfigPath, 'utf8'));
  if (config.schemaVersion !== 1) {
    throw new Error(`AUTHOR_PACKAGE_SCHEMA_UNSUPPORTED: ${config.schemaVersion}`);
  }
  if (!config.id || typeof config.id !== 'string') {
    throw new Error('AUTHOR_PACKAGE_ID_MISSING: deck.config.json must define id');
  }
  if (!config.entry || typeof config.entry !== 'string') {
    throw new Error('AUTHOR_PACKAGE_ENTRY_MISSING: deck.config.json must define entry');
  }

  const styleFiles = (config.styles || []).map((relativePath) => {
    const filePath = resolveInside(rootDir, relativePath, 'AUTHOR_STYLE_FILE_OUTSIDE_ROOT');
    if (!fs.existsSync(filePath)) throw new Error(`AUTHOR_STYLE_FILE_MISSING: ${relativePath}`);
    return { relativePath: slash(relativePath), filePath };
  });

  const pageFiles = (config.pages || []).map((page) => {
    if (!page || !page.id || !page.file) {
      throw new Error('AUTHOR_PAGE_ENTRY_INVALID: each page entry must define id and file');
    }
    const filePath = resolveInside(rootDir, page.file, 'AUTHOR_PAGE_FILE_OUTSIDE_ROOT');
    if (!fs.existsSync(filePath)) throw new Error(`AUTHOR_PAGE_FILE_MISSING: ${page.file}`);
    return { id: page.id, relativePath: slash(page.file), filePath };
  });

  if (pageFiles.length === 0) {
    throw new Error('AUTHOR_PAGE_LIST_EMPTY: deck.config.json must define at least one page');
  }

  return {
    rootDir,
    configPath: resolvedConfigPath,
    config,
    entryPath: resolveInside(rootDir, config.entry, 'AUTHOR_ENTRY_OUTSIDE_ROOT'),
    styleFiles,
    pageFiles,
  };
}

function assembleAuthorPackage(configPath) {
  const sourcePackage = readAuthorPackage(configPath);
  const styles = sourcePackage.styleFiles
    .map((file) => {
      const css = fs.readFileSync(file.filePath, 'utf8');
      return `<style data-source-file="${attr(file.relativePath)}">\n${css.trimEnd()}\n</style>`;
    })
    .join('\n');

  const pages = sourcePackage.pageFiles
    .map((file) => addSourceFileAttribute(fs.readFileSync(file.filePath, 'utf8'), file.relativePath))
    .join('\n\n');

  const title = sourcePackage.config.title || sourcePackage.config.id;
  const profileAttr = sourcePackage.config.profile ? ` data-id-profile="${attr(sourcePackage.config.profile)}"` : '';
  const html = [
    '<!doctype html>',
    '<html lang="zh-CN">',
    '<head>',
    '  <meta charset="utf-8">',
    `  <title>${escapeText(title)}</title>`,
    `  ${GENERATED_COMMENT}`,
    styles,
    '</head>',
    '<body>',
    `  <main class="deck" data-id-document="${attr(sourcePackage.config.id)}"${profileAttr}>`,
    indent(pages, 4),
    '  </main>',
    '</body>',
    '</html>',
    '',
  ].join('\n');

  return { html, sourcePackage };
}

function writeAuthorPackageEntry(configPath) {
  const { html, sourcePackage } = assembleAuthorPackage(configPath);
  fs.writeFileSync(sourcePackage.entryPath, html, 'utf8');
  return { entryPath: sourcePackage.entryPath };
}

function checkAuthorPackageEntry(configPath) {
  const { html, sourcePackage } = assembleAuthorPackage(configPath);
  if (!fs.existsSync(sourcePackage.entryPath)) {
    return {
      ok: false,
      entryPath: sourcePackage.entryPath,
      message: 'generated deck.html is missing',
    };
  }
  const current = fs.readFileSync(sourcePackage.entryPath, 'utf8');
  if (current !== html) {
    return {
      ok: false,
      entryPath: sourcePackage.entryPath,
      message: 'generated deck.html is out of date',
    };
  }
  return { ok: true, entryPath: sourcePackage.entryPath };
}

function addSourceFileAttribute(fragment, sourceFile) {
  return fragment.replace(/<section\b([^>]*)>/i, (match, rest) => {
    if (/data-id-source-file\s*=/.test(rest)) return match;
    return `<section${rest} data-id-source-file="${attr(sourceFile)}">`;
  }).trimEnd();
}

function resolveInside(rootDir, relativePath, errorCode) {
  const target = path.resolve(rootDir, relativePath);
  const root = path.resolve(rootDir);
  const rootWithSeparator = root.endsWith(path.sep) ? root : `${root}${path.sep}`;
  if (target !== root && !target.startsWith(rootWithSeparator)) {
    throw new Error(`${errorCode}: ${relativePath}`);
  }
  return target;
}

function indent(value, spaces) {
  const prefix = ' '.repeat(spaces);
  return String(value).split(/\r?\n/).map((line) => line ? `${prefix}${line}` : line).join('\n');
}

function slash(value) {
  return String(value).replace(/\\/g, '/');
}

function attr(value) {
  return String(value).replace(/&/g, '&amp;').replace(/"/g, '&quot;').replace(/</g, '&lt;');
}

function escapeText(value) {
  return String(value).replace(/&/g, '&amp;').replace(/</g, '&lt;');
}

module.exports = {
  GENERATED_COMMENT,
  assembleAuthorPackage,
  checkAuthorPackageEntry,
  readAuthorPackage,
  writeAuthorPackageEntry,
};
```

Create `scripts/assemble-authoring.js`:

```js
#!/usr/bin/env node

const path = require('path');
const { checkAuthorPackageEntry, writeAuthorPackageEntry } = require('../src/authoring');

main();

function main() {
  try {
    const options = parseArgs(process.argv.slice(2));
    if (options.help || !options.packagePath) {
      printUsage(options.packagePath ? 0 : 1);
      return;
    }

    const configPath = path.resolve(options.packagePath);
    if (options.check) {
      const result = checkAuthorPackageEntry(configPath);
      if (!result.ok) {
        console.error(`${result.message}: ${result.entryPath}`);
        process.exit(1);
      }
      console.log(`Authoring package is up to date: ${result.entryPath}`);
      return;
    }

    const result = writeAuthorPackageEntry(configPath);
    console.log(`Wrote ${result.entryPath}`);
  } catch (error) {
    console.error(error && error.stack ? error.stack : String(error));
    process.exit(1);
  }
}

function parseArgs(args) {
  const out = { packagePath: null, check: false, help: false };
  for (let index = 0; index < args.length; index += 1) {
    const arg = args[index];
    if (arg === '--') continue;
    if (arg === '--help' || arg === '-h') out.help = true;
    else if (arg === '--check') out.check = true;
    else if (arg === '--package') out.packagePath = args[index += 1];
    else if (arg.startsWith('--package=')) out.packagePath = arg.slice('--package='.length);
    else if (!out.packagePath) out.packagePath = arg;
    else throw new Error(`Unknown argument: ${arg}`);
  }
  return out;
}

function printUsage(exitCode) {
  const usage = [
    'Usage: npm run assemble:authoring -- --package <deck.config.json> [--check]',
    '',
    'Options:',
    '  --package <file>  Authoring source package config.',
    '  --check           Verify generated deck.html is up to date.',
  ].join('\n');
  const writer = exitCode ? console.error : console.log;
  writer(usage);
  process.exit(exitCode);
}
```
